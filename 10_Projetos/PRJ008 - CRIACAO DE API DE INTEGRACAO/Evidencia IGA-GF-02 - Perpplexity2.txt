#!/bin/bash
# =================================================================
# PRJ003 GMUD-011 - Deploy midPoint 4.9 Manual Control (Ubuntu 24.04)
# Executor: Paulo Feitosa | Data: 21/01/2026
# =================================================================

set -e

# CONFIGURAÇÃO
PRJ_PATH="/srv/iga-project"
COMPOSE_FILE="docker-compose.yml"
SQL_URLS=(
    "https://raw.githubusercontent.com/Evolveum/midpoint/v4.9/config/sql/native-new/postgres.sql"
    "https://raw.githubusercontent.com/Evolveum/midpoint/v4.9/config/sql/native-new/postgres-audit.sql"
    "https://raw.githubusercontent.com/Evolveum/midpoint/v4.9/config/sql/native-new/postgres-quartz.sql"
)

echo "🚀 PRJ003 GMUD-011 | midPoint 4.9 Greenfield Deploy"
echo "📍 VM: $(hostname -I | awk '{print $1}')"
echo "⏰ $(date)"
echo "----------------------------------------"

# PASSO 1: LIMPEZA NUCLEAR (Estado Virginal)
echo "🧹 PASSO 1: LIMPEZA NUCLEAR"
cd $PRJ_PATH || { echo "❌ Diretório $PRJ_PATH não existe"; exit 1; }
sudo docker compose -f $COMPOSE_FILE down -v 2>/dev/null || true
sudo rm -rf data *.sql config
mkdir -p data/{postgres,midpoint,var}
echo "✅ Ambiente limpo - estado virgem"

# PASSO 2: DOCKER-COMPOSE (GMUD-011 Validado)
cat > $COMPOSE_FILE << 'EOF'
version: '3.8'
services:
  postgres:
    image: postgres:16
    container_name: iga-postgres
    environment:
      POSTGRES_DB: midpoint
      POSTGRES_USER: midpointuser
      POSTGRES_PASSWORD: P0stgr3sS3cur32026!
    volumes:
      - ./data/postgres:/var/lib/postgresql/data
    networks:
      - iga-network
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U midpointuser -d midpoint"]
      interval: 5s
      timeout: 5s
      retries: 5
      start_period: 10s

  midpoint:
    image: evolveum/midpoint:4.9
    container_name: iga-midpoint
    ports:
      - "8080:8080"
    volumes:
      - ./data/midpoint/var:/opt/midpoint/var
    depends_on:
      postgres:
        condition: service_healthy
    environment:
      REPODATABASETYPE: postgresql
      REPOJDBCURL: jdbc:postgresql://postgres:5432/midpoint?user=midpointuser&password=P0stgr3sS3cur32026!
      JAVAOPTS: -Xms512m -Xmx1024m -Dmidpoint.repository.database=postgresql -Dfile.encoding=UTF-8
      MPKEYSTOREPASSWORD: midpointkeystore2026
      MPSET_midpoint_administrator_initialPassword: M1dP0!ntAdm!n2026
    networks:
      - iga-network

networks:
  iga-network:
    driver: bridge
EOF
echo "✅ docker-compose.yml criado"

# PASSO 3: POSTGRESQL
echo "🐘 PASSO 2: POSTGRESQL 16"
sudo docker compose up -d postgres
echo "⏳ Aguardando healthcheck (30s)..."
sleep 30

if sudo docker exec iga-postgres pg_isready -U midpointuser -d midpoint 2>&1 | grep -q "accepting"; then
    echo "✅ PostgreSQL 16 READY"
else
    echo "❌ POSTGRESQL FAILED"; sudo docker logs iga-postgres --tail 20; exit 1
fi

# PASSO 4: SCHEMA INJEÇÃO MANUAL (SQALE nativo)
echo "📋 PASSO 3: SCHEMA midPoint 4.9 (SQALE)"
for url in "${SQL_URLS[@]}"; do
    file=$(basename "$url")
    echo "📥 $file"
    wget -q "$url" -O "$file"
    sudo docker exec -i iga-postgres psql -U midpointuser -d midpoint < "$file"
    rm "$file"
done

# Validação Schema (85+ tabelas)
tables=$(sudo docker exec iga-postgres psql -U midpointuser -d midpoint -t -c \
    "SELECT COUNT(*) FROM information_schema.tables WHERE table_schema='public'" | tr -d ' ')
[[ $tables -ge 85 ]] && echo "✅ Schema OK: $tables tabelas" || { echo "❌ Schema incompleto: $tables"; exit 1; }

# PASSO 5: MIDPOINT BOOT
echo "⚙️ PASSO 4: midPoint 4.9 BOOT"
sudo docker compose up -d midpoint
echo "⏳ Inicialização (90s)..."
sleep 90

# PASSO 6: VALIDAÇÃO CRÍTICA
echo "🔍 PASSO 5: VALIDAÇÃO"
logs=$(sudo docker logs iga-midpoint 2>&1 | tail -100)

# Verificações obrigatórias GMUD-011
if echo "$logs" | grep -q "Started MidPointSpringApplication"; then
    echo "✅ [1/4] Spring Boot UP"
else
    echo "⚠️ Spring Boot ainda inicializando"
fi

if echo "$logs" | grep -q "Database schema is compliant"; then
    echo "✅ [2/4] Schema VALIDADO"
else
    echo "❌ [2/4] Schema NÃO validado"
    echo "=== LOGS PROBLEMÁTICOS ==="
    echo "$logs" | tail -20
    exit 1
fi

http_code=$(sudo docker exec iga-midpoint curl -s -o /dev/null -w "%{http_code}" http://localhost:8080/midpoint 2>/dev/null || echo "000")
[[ $http_code = "200" ]] && echo "✅ [3/4] HTTP 200 OK" || echo "⚠️ [3/4] HTTP $http_code"

sudo docker ps --format "table {{.Names}}\t{{.Status}}" | grep iga

# SUCESSO 🎉
cat << EOF

🎉✅ GMUD-011 EXECUTADA COM SUCESSO ✅🎉

🌐 ACESSO:
   URL: http://$(hostname -I | awk '{print $1}'):8080/midpoint

👤 CREDENCIAIS:

📊 MONITORAMENTO:

✅ CRITÉRIOS ATINGIDOS GMUD-011:
   [x] PostgreSQL 16 + Schema SQALE (85+ tabelas)
   [x] midPoint 4.9 UP + "Database schema is compliant"
   [x] HTTP 200 /midpoint
   [x] Credenciais funcionais

EOF

echo "💾 Volumes persistentes em: $PRJ_PATH/data"
echo "🔄 Para restart: sudo docker compose restart"
echo "🛑 Para parar: sudo docker compose down"
