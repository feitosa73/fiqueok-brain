import hvac
import os
import sys

class VaultClient:
    def __init__(self):
        # Configurações via variáveis de ambiente (PAD-002)
        self.vault_url = os.getenv("VAULT_ADDR", "http://xxx.xxx.xxx.xxx:8200")
        self.role_id = os.getenv("VAULT_ROLE_ID")
        self.secret_id = os.getenv("VAULT_SECRET_ID")
        self.client = hvac.Client(url=self.vault_url)

    def get_anythingllm_key(self):
        try:
            # Autenticação via AppRole
            auth_result = self.client.auth.approle.login(
                role_id=self.role_id,
                secret_id=self.secret_id
            )

            # Recupera o segredo no path definido na política
            read_response = self.client.secrets.kv.v2.read_secret_version(
                path='anythingllm',
                mount_point='secret'
            )

            return read_response['data']['data']['api_key']

        except Exception as e:
            print(f"CRITICAL: Falha na autenticação Vault ou leitura de segredo: {e}")
            sys.exit(1)

if __name__ == "__main__":
    # Teste de conectividade simples
    vc = VaultClient()
    print("Conexão com Vault estabelecida com sucesso.")

import json
from datetime import datetime

class JsonFormatter(logging.Formatter):
    def format(self, record):
        log_record = {
            "timestamp": datetime.utcnow().isoformat(),
            "level": record.levelname,
            "message": record.getMessage(),
            "module": record.module,
            "prj_id": "PRJ019"
        }
        # Adiciona campos extras se presentes (ex: hash SHA256, workspace)
        if hasattr(record, 'extra_info'):
            log_record.update(record.extra_info)

        return json.dumps(log_record)

def setup_logger():
    logger = logging.getLogger("prj019_audit")
    logger.setLevel(logging.INFO)

    # Handler para console (Docker logs)
    handler = logging.StreamHandler()
    handler.setFormatter(JsonFormatter())
    logger.addHandler(handler)

    return logger

import hvac
import os
import sys

class VaultClient:
    def __init__(self):
        self.vault_url = os.getenv("VAULT_ADDR", "http://xxx.xxx.xxx.xxx:8200")
        # Caminho do token gerado pelo Vault Agent (Sink File)
        self.token_path = "/app/token/.vault-token"
        self.client = hvac.Client(url=self.vault_url)

    def get_anythingllm_key(self):
        try:
            # Lendo o token do arquivo (Gerenciado pelo Agent)
            with open(self.token_path, 'r') as f:
                self.client.token = f.read().strip()

            # Busca o segredo
            read_response = self.client.secrets.kv.v2.read_secret_version(
                path='anythingllm',
                mount_point='secret'
            )
            return read_response['data']['data']['api_key']

        except Exception as e:
            print(f"CRITICAL: Erro ao ler token do Agent ou segredo do Vault: {e}")
            sys.exit(1)

def pre_flight_check():
    audit_log.info("Iniciando Verificação de Conectividade (Pre-flight)...")
    vc = VaultClient()

    try:
        # Tenta obter a chave do AnythingLLM via Vault Agent (Sidecar)
        api_key = vc.get_anythingllm_key()
        if api_key:
            audit_log.info("CONEXÃO SUCESSO: Token lido do Agent e Segredo recuperado do Vault.")
            return True
    except Exception as e:
        audit_log.error(f"FALHA NO PRE-FLIGHT: Certifique-se que o Vault Agent está rodando. Erro: {e}")
        return False

if __name__ == "__main__":
    # Aguarda o Agent inicializar (pequeno delay de segurança)
    time.sleep(5)

    if pre_flight_check():
        audit_log.info("Sistema operacional. Monitorando /data/obsidian...")
        while True:
            # Mantém o container vivo para o monitoramento
            time.sleep(60)
    else:
        audit_log.critical("Falha crítica na autenticação inicial. Encerrando.")
        sys.exit(1)
